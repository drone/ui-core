import Toggle, {ToggleItem} from "../../src/components/toggle";
import Playground from "../../docs/playground";
import { useState } from "react";

# Default

The default toggle comonent:

<Playground
  scope={{ Toggle, ToggleItem }}
  code={`() => (
    <Toggle defaultValue="Yaml">
        <ToggleItem text="Yaml" />
        <ToggleItem text="Diff" />
    </Toggle>
);`}
/>

The toggle component can be used to toggle between multiple items:

<Playground
  scope={{ Toggle, ToggleItem, useState }}
  code={`() => {
  const [ value, setValue ] = useState("Yaml");
  return (
    <Toggle value={value} onValueChange={(value) => setValue(value)}>
        <ToggleItem text="Yaml" />
        <ToggleItem text="Json" />
        <ToggleItem text="Xml" />
    </Toggle>
  );
};`}
/>
