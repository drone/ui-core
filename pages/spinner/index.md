import Spinner from "../../src/components/Spinner";
import Playground from "../../docs/playground";

# Default

The default form of a spinner:

<Playground
  scope={{ Spinner }}
  code={`<Spinner />`}
/>

The spinner with a custom color:

<Playground
  scope={{ Spinner }}
  code={`<Spinner size="64px" />`}
/>

The spinner with a custom color:

<Playground
  scope={{ Spinner }}
  code={`<Spinner size="64px" color="#0092e4" />`}
/>
