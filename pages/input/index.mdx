import Input from "../../src/components/input";
import Playground from "../../docs/playground";

# Default

The default form of a text field.

<Playground
  scope={{ Input }}
  code={`<Input placeholder="Username" />`}
/>

Use the type attribute to change the input type:

<Playground
  scope={{ Input }}
  code={`<Input placeholder="Password" type="password" />`}
/>

# Disabled

Use the disabled attribute to disable the input:

<Playground
  scope={{ Input }}
  code={`<Input placeholder="Username" disabled={true} />`}
/>
