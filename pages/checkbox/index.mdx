import { Checkbox } from "../../src/components/checkbox/Checkbox";
import Playground from "../../docs/playground";
import { useState } from "react";

# Default

By default, a checkbox input includes a selected and unselected state.

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox />`}
/>

# Checked

The checkbox state is managed by react and is set to checked:

<Playground
  scope={{ Checkbox, useState }}
  code={`() => {
    const [ checked, setChecked ] = useState(true);
    return <Checkbox onCheckedChange={() => setChecked(!checked)} checked={checked} />
};`}
/>

# Unchecked

The checkbox state is managed by react and is set to unchecked:

<Playground
  scope={{ Checkbox, useState }}
  code={`() => {
    const [ checked, setChecked ] = useState(false);
    return <Checkbox onCheckedChange={() => setChecked(!checked)} checked={checked} />
};`}
/>


# Disabled

The checkbox is disabled:

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox disabled={true} />`}
/>

The checkbox state is disabled, and the checkbox is checked:

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox disabled={true} checked={true} />`}
/>
