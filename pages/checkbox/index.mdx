import Checkbox from "../../src/components/Checkbox";
import Playground from "../../docs/playground";

# Default

The default checkbox.

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox />`}
/>

# Checked

The checkbox state is managed by react and is set to checked:

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox checked={true} />`}
/>

# Unchecked

The checkbox state is managed by react and is set to unchecked:

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox checked={false} />`}
/>


# Disabled

The checkbox is disabled:

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox disabled={true} />`}
/>

The checkbox state is disabled, and the checkbox is checked:

<Playground
  scope={{ Checkbox }}
  code={`<Checkbox disabled={true} checked={true} />`}
/>
